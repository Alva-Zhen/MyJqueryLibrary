import React from 'react';

/**
 * Slide 轮播图
 * @param {object} data - 图片对象.
 * @param {number} data.width - 宽度.
 * @param {number} data.height - 高度.
 * @param {Array} data.img - 图片数组.
 * @param {number} data.time - 切换时间.
 * @desc 轮播图
 * @example <Slide data={{
    width: 653,
    height: 422,
    img: this.state.bannerImg,
    time: 5000,
}} />
 */
class Slide extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            img: props.data.img || '',
            currImgIndex: 1,
            arrow: false
        };
    }

    handleTime() {
        this.interval = setInterval(() => {
            let state = this.state,
                img = state.img,
                imgLength = img.length,
                currImgIndex = state.currImgIndex;
            if (currImgIndex < imgLength) {
                this.setState({currImgIndex: currImgIndex + 1})
            } else {
                this.setState({currImgIndex: 1})
            }
        }, this.props.data.time)
    }

    handleArrowFun(type) {
        let state = this.state,
            currImgIndex = state.currImgIndex,
            imgLength = state.img.length;

        if (type) {
            if (currImgIndex === imgLength) {
                this.setState({currImgIndex: 1})
            } else {
                this.setState({currImgIndex: currImgIndex + 1})
            }

        } else {
            if (currImgIndex === 1) {
                this.setState({currImgIndex: imgLength})
            } else {
                this.setState({currImgIndex: currImgIndex - 1})
            }
        }
    }

    componentDidMount() {
        if (this.state.img.length > 1) {
            this.handleTime()
        }
    }

    componentWillUnmount() {
        this.interval && clearInterval(this.interval);
    }

    render() {
        let state = this.state, img = state.img, currImgIndex = state.currImgIndex;
        return (
            <div>
                {
                    img ?
                        <div className="common_slide"
                             style={{width: this.props.data.width, height: this.props.data.height}}
                             onMouseEnter={() => {
                                 this.setState({arrow: true})
                             }}
                             onMouseLeave={() => {
                                 this.setState({arrow: false})
                             }}
                        >
                            {
                                img.map((data, index) => {
                                    return (
                                        <a href={data.source} key={index} target="_blank">
                                            <img
                                                className={(currImgIndex === index + 1) ? "animation_slide_fade_image" : "none"}
                                                src={data.img} alt=""/>
                                        </a>
                                    )
                                })
                            }

                            {
                                img.length > 1 ?
                                    <div className="buttonDiv">
                                        <div className="button" style={{width: img.length * 24 + 32}}>
                                            {
                                                img.map((data, index) => {
                                                    return <button key={index}
                                                                   className={(currImgIndex === index + 1) ? "on" : ""}
                                                                   onMouseEnter={() => {
                                                                       this.setState({currImgIndex: index + 1})
                                                                   }}
                                                    >
                                                        {/* {index + 1}*/}</button>
                                                })
                                            }
                                        </div>
                                    </div>
                                    :
                                    null
                            }
                            {
                                state.arrow && img.length > 1 ?
                                    <div className="arrow">
                                        <i className="ico-slide-arrow-left"
                                           onClick={this.handleArrowFun.bind(this, 0)}/>
                                        <i className="ico-slide-arrow-right"
                                           onClick={this.handleArrowFun.bind(this, 1)}/>
                                    </div>
                                    :
                                    null
                            }
                        </div>
                        :
                        null
                }
            </div>
        );
    }
}

export default Slide;
